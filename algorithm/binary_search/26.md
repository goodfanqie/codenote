#include <vector>
using namespace std;
// 我的答案:其实就是硬生生想出来的：
// 找到一个中间数，一个记录修改后的数组的元素，一个j迭代数
// 就是互相比较一样他就跳过，不一样中间数就重新赋值比较，同时i增加
class Solution {
public:
int removeDuplicates(vector<int>& nums) {
    int i = 0;
    for (int temp = nums[0], j = 1; j < nums.size(); ++j) {
        if (temp != nums[j]) {
            temp = nums[j];
            nums[++i] = nums[j];
        }
    }
    return ++i;
}
};
